bool solve(string s, string p){
    int s_len = s.length(), p_len = p.length();
    vector<vector<bool>> dp(s_len + 1, vector<bool>(p_len + 1, false));
    dp[0][0] = true;
    for(int j = 1; j <= p_len; ++j){
        if(p[j - 1] == '*'){
            dp[0][j] = dp[0][j-1];
        }
        else{
            dp[0][j] = false;
        }
    }
    for(int i = 1; i <= s_len; ++i){
        for(int j = 1; j <= p_len; ++j){
            if(p[j-1] == '*'){
                dp[i][j] = dp[i-1][j] || dp[i][j-1] || dp[i-1][j-1];
            }
            else{
                if(p[j-1] == s[i-1]){
                    dp[i][j] = dp[i-1][j-1];
                }
                else{
                    dp[i][j] = false;
                }
            }
        }
    }
    return dp[s_len][p_len];
}

int main()
{

    int n = 0;
    string p, s;
    cin >> p;
    cin >> n;
    vector<string> l;
    for(int i = 0; i < n; i++){
        cin >> s;
        if(solve(s, p)){
            l.push_back(s);
        }
    }
    for(string s : l){
        cout << s << endl;
    }
    return 0;
}
