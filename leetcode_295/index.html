from sortedcontainers import SortedList
class MedianFinder:

    def __init__(self):
        """
        initialize your data structure here.
        """
        self.sl = SortedList()
        

    def addNum(self, num: int) -> None:
        self.sl.add(num)

    def findMedian(self) -> float:
        l = len(self.sl)
        if l % 2 == 0:
            return (self.sl[l//2] + self.sl[l//2-1]) / 2
        else:
            return self.sl[l//2]
